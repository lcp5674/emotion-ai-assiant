## 二、备份执行脚本

### 2.1 MySQL全量备份脚本

```bash
#!/bin/bash
# backup_mysql.sh - MySQL数据库备份脚本

# 配置
BACKUP_DIR="/backup/mysql"
DATE=$(date +%Y%m%d_%H%M%S)
DB_HOST="localhost"
DB_USER="emotion_ai"
DB_PASS="${MYSQL_PASSWORD}"
DB_NAME="emotion_ai"
RETENTION_DAYS=30

# 创建备份目录
mkdir -p ${BACKUP_DIR}

# 执行备份
echo "Starting MySQL backup at $(date)"

mysqldump -h${DB_HOST} -u${DB_USER} -p${DB_PASS} \
    --single-transaction \
    --routines \
    --triggers \
    --events \
    --master-data=2 \
    ${DB_NAME} > ${BACKUP_DIR}/full_backup_${DATE}.sql

# 压缩备份文件
gzip ${BACKUP_DIR}/full_backup_${DATE}.sql

# 清理过期备份
find ${BACKUP_DIR} -name "full_backup_*.sql.gz" -mtime +${RETENTION_DAYS} -delete

echo "Backup completed at $(date)"
echo "Backup file: ${BACKUP_DIR}/full_backup_${DATE}.sql.gz"
```

### 2.2 备份计划 (crontab)

```bash
# 每天凌晨2点执行全量备份
0 2 * * * /opt/backup/backup_mysql.sh >> /var/log/backup/mysql.log 2>&1

# 每6小时执行增量备份
0 */6 * * * /opt/backup/backup_incremental.sh >> /var/log/backup/incremental.log 2>&1

# 每周日凌晨3点执行完整备份（额外保险）
0 3 * * 0 /opt/backup/backup_full_weekly.sh >> /var/log/backup/weekly.log 2>&1
```

## 三、恢复流程

### 3.1 数据库恢复脚本

```bash
#!/bin/bash
# restore_mysql.sh - 数据库恢复脚本

# 配置
BACKUP_FILE=$1
DB_HOST="localhost"
DB_USER="emotion_ai"
DB_PASS="${MYSQL_PASSWORD}"
DB_NAME="emotion_ai"

if [ -z "$BACKUP_FILE" ]; then
    echo "Usage: $0 <backup_file.sql.gz>"
    exit 1
fi

if [ ! -f "$BACKUP_FILE" ]; then
    echo "Error: Backup file not found: $BACKUP_FILE"
    exit 1
fi

echo "WARNING: This will overwrite the current database!"
echo "Press Ctrl+C to cancel, or Enter to continue..."
read

echo "Starting database restore at $(date)"

# 解压并恢复
gunzip -c ${BACKUP_FILE} | mysql -h${DB_HOST} -u${DB_USER} -p${DB_PASS} ${DB_NAME}

if [ $? -eq 0 ]; then
    echo "Database restore completed successfully at $(date)"
else
    echo "ERROR: Database restore failed!"
    exit 1
fi
```

### 3.2 恢复验证

```bash
# 验证数据完整性
mysql -u${DB_USER} -p${DB_PASSWORD} -e "SELECT COUNT(*) FROM users;" ${DB_NAME}
mysql -u${DB_USER} -p${DB_PASSWORD} -e "SELECT COUNT(*) FROM conversations;" ${DB_NAME}
mysql -u${DB_USER} -p${DB_PASSWORD} -e "SELECT COUNT(*) FROM emotion_diaries;" ${DB_NAME}
```

```bash
#!/bin/bash
# verify_backup.sh - 备份健康检查

BACKUP_DIR="/backup/mysql"
LAST_BACKUP=$(find ${BACKUP_DIR} -name "*.sql.gz" -type f -mmin -60 | head -1)

if [ -z "$LAST_BACKUP" ]; then
    echo "WARNING: No backup in the last 60 minutes!"
    # 发送告警
    exit 1
fi

echo "Last backup: $LAST_BACKUP"
```
