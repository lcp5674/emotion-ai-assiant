## 四、部署脚本 (deploy.sh)

### 4.1 创建完整部署脚本

```bash
#!/bin/bash
# emotion-ai-assiant 一键部署脚本

set -e

echo "=========================================="
echo "  心灵伴侣AI - 一键部署脚本"
echo "=========================================="

# 变量配置
PROJECT_DIR="/opt/emotion-ai-assiant"
BACKUP_DIR="/opt/emotion-ai-backup"
SERVER_IP="118.25.110.184"

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

log_info() { echo -e "${GREEN}[INFO]${NC} $1"; }
log_warn() { echo -e "${YELLOW}[WARN]${NC} $1"; }
log_error() { echo -e "${RED}[ERROR]${NC} $1"; }

# 1. 检查Docker
check_docker() {
    log_info "检查Docker环境..."
    if ! command -v docker &> /dev/null; then
        log_error "Docker未安装，正在安装..."
        curl -fsSL https://get.docker.com | sh
        systemctl start docker
        systemctl enable docker
    fi
    docker --version
    docker-compose --version
}

# 2. 克隆/更新代码
deploy_code() {
    log_info "部署代码..."
    if [ ! -d "$PROJECT_DIR" ]; then
        git clone https://github.com/lcp5674/emotion-ai-assiant.git $PROJECT_DIR
    else
        cd $PROJECT_DIR && git pull
    fi
}

# 3. 创建环境文件
create_env() {
    log_info "创建环境配置文件..."
    cat > $PROJECT_DIR/.env << 'ENVEOF'
APP_NAME=心灵伴侣AI
APP_VERSION=1.0.0
DEBUG=false
HOST=0.0.0.0
PORT=8000

MYSQL_HOST=mysql
MYSQL_PORT=3306
MYSQL_USER=emotion_ai
MYSQL_PASSWORD=EmotionAi2024!
MYSQL_DATABASE=emotion_ai

REDIS_HOST=redis
REDIS_PORT=6379

SECRET_KEY=production-secret-key-change-this-in-production-12345
ALGORITHM=HS256
ACCESS_TOKEN_EXPIRE_MINUTES=10080

LLM_PROVIDER=mock
VECTOR_DB_TYPE=memory

MINIO_ENDPOINT=minio:9000
MINIO_ACCESS_KEY=minioadmin
MINIO_SECRET_KEY=minioadmin123

CORS_ORIGINS=["http://SERVER_IP"]
ENVEOF
}

# 4. 启动服务
start_services() {
    log_info "启动Docker服务..."
    cd $PROJECT_DIR
    docker-compose down 2>/dev/null || true
    docker-compose up -d --build
    
    log_info "等待服务启动..."
    sleep 30
}

# 5. 检查状态
check_status() {
    log_info "检查服务状态..."
    docker-compose ps
    
    log_info "检查后端健康..."
    curl -s http://localhost:8000/health || log_warn "后端未响应"
    
    log_info "检查Nginx..."
    curl -s -I http://localhost || log_warn "前端未响应"
}

# 执行部署
main() {
    check_docker
    deploy_code
    create_env
    start_services
    check_status
    
    log_info "=========================================="
    log_info "部署完成！"
    log_info "访问地址: http://$SERVER_IP"
    log_info "API文档: http://$SERVER_IP/docs"
    log_info "=========================================="
}

main "$@"
```

## 五、Docker Compose 配置

### 5.1 docker-compose.prod.yml (推荐 - 无端口冲突)

```bash
# 使用 prod 版本避免端口冲突
docker-compose -f docker-compose.prod.yml up -d --build
```

### 5.2 docker-compose.yml

```yaml
version: '3.8'

services:
  # MySQL数据库
  mysql:
    image: mysql:8.0
    container_name: emotion-ai-mysql
    restart: always
    environment:
      MYSQL_ROOT_PASSWORD: ${MYSQL_ROOT_PASSWORD:-root_password}
      MYSQL_DATABASE: ${MYSQL_DATABASE:-emotion_ai}
      MYSQL_USER: ${MYSQL_USER:-emotion_ai}
      MYSQL_PASSWORD: ${MYSQL_PASSWORD:-emotion_ai_password}
    ports:
      - "${MYSQL_PORT:-3306}:3306"
    volumes:
      - mysql_data:/var/lib/mysql
    command: --default-authentication-plugin=mysql_native_password --character-set-server=utf8mb4 --collation-server=utf8mb4_unicode_ci
    healthcheck:
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Redis缓存
  redis:
    image: redis:7-alpine
    container_name: emotion-ai-redis
    restart: always
    ports:
      - "${REDIS_PORT:-6379}:6379"
    volumes:
      - redis_data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  # MinIO对象存储
  minio:
    image: minio/minio:latest
    container_name: emotion-ai-minio
    restart: always
    ports:
      - "${MINIO_PORT:-9000}:9000"
      - "${MINIO_CONSOLE_PORT:-9001}:9001"
    environment:
      MINIO_ROOT_USER: ${MINIO_ROOT_USER:-minioadmin}
      MINIO_ROOT_PASSWORD: ${MINIO_ROOT_PASSWORD:-minioadmin123}
    volumes:
      - minio_data:/data
    command: server /data --console-address ":9001"
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:9000/minio/health/live"]
      interval: 30s
      timeout: 20s
      retries: 3

  # 后端API服务
  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: emotion-ai-backend
    restart: always
    ports:
      - "${BACKEND_PORT:-8000}:8000"
    environment:
      - MYSQL_HOST=mysql
      - MYSQL_PORT=3306
      - MYSQL_USER=${MYSQL_USER:-emotion_ai}
      - MYSQL_PASSWORD=${MYSQL_PASSWORD:-emotion_ai_password}
      - MYSQL_DATABASE=${MYSQL_DATABASE:-emotion_ai}
      - REDIS_HOST=redis
      - REDIS_PORT=6379
      - LLM_PROVIDER=${LLM_PROVIDER:-mock}
      - SECRET_KEY=${SECRET_KEY:-change-me-in-production}
      - ENVIRONMENT=production
    depends_on:
      mysql:
        condition: service_healthy
      redis:
        condition: service_healthy
      minio:
        condition: service_healthy
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 60s

  # Nginx反向代理
  nginx:
    image: nginx:alpine
    container_name: emotion-ai-nginx
    restart: always
    ports:
      - "${HTTP_PORT:-80}:80"
      - "${HTTPS_PORT:-443}:443"
    volumes:
      - ./nginx.conf:/etc/nginx/conf.d/default.conf:ro
      - ./frontend/web/dist:/usr/share/nginx/html:ro
      - nginx_logs:/var/log/nginx
    depends_on:
      backend:
        condition: service_healthy
    healthcheck:
      test: ["CMD", "nginx", "-t"]
      interval: 30s
      timeout: 10s
      retries: 3

volumes:
  mysql_data:
  redis_data:
  minio_data:
  nginx_logs:

networks:
  default:
    name: emotion-ai-network
```

## 六、Nginx 配置

### nginx.conf

```nginx
server {
    listen 80;
    server_name _;
    
    # 前端静态文件
    location / {
        root /usr/share/nginx/html;
        index index.html;
        try_files $uri $uri/ /index.html;
    }
    
    # API代理
    location /api/ {
        proxy_pass http://backend:8000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_read_timeout 300s;
        proxy_connect_timeout 75s;
    }
    
    # WebSocket代理
    location /ws/ {
        proxy_pass http://backend:8000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_read_timeout 86400s;
    }
    
    # 静态资源缓存
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
        root /usr/share/nginx/html;
        expires 1y;
        add_header Cache-Control "public, immutable";
    }
}
```

## 十一、故障排查

### 11.1 常见问题

| 问题 | 解决方案 |
|------|----------|
| 后端启动失败 | 检查.env配置，查看 `docker-compose logs backend` |
| 数据库连接失败 | 等待MySQL完全启动，检查 `docker-compose logs mysql` |
| 前端无法访问 | 检查Nginx配置和容器状态 |
| 端口被占用 | 修改.env中的端口或停止占用进程 |

### 11.2 日志查看
